\def\test#1#2{%
  \def\@temp{#1}%
  \ifx\@temp\@empty
    \makebox[5.5cm][l]{\texttt{\LARGE pagebox is absent}}\fbox{\includegraphics{#2.pdf}}%
  \else
    \makebox[5.5cm][l]{\texttt{\LARGE pagebox=#1box}}\fbox{\includegraphics[pagebox=#1box]{#2.pdf}}%
  \fi
  \par
}
\makeatother
